from flask import Flask, render_template from datetime import datetime app = Flask(__name__) @app.route('/') def home(): # ข้อมูลที่เราต้องการส่งไปแสดงในหน้า HTML name = "สมชาย" now = datetime.now().strftime("%H:%M:%S") # ส่งตัวแปรไปยัง index.html return render_template('index.html', user_name=name, current_time=now) if __name__ == '__main__': app.run(debug=True)
